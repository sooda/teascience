\section{Menetelmä}
\subsection*{}

\begin{frame}
	\frametitle{Tutkimuksen menetelmä}
	\begin{itemize}
	\item Sopiva muki
	\item Sopiva tee, riittävän konsistentti ettei siitäkin tule ylimääräistä muuttujaa
	\item Vesi keittimestä mukiin ja teesihti sekaan
	\item Haudutellaan teelaadun ohjeen mukaan
	\item Hörpitään vähän kerrallaan, merkitään mittaukset muistiin
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Koejärjestelyn alustus}
	\begin{itemize}
	\item Musta tee "sadassa" celsiusasteessa
	\item Vihreä tee 75 asteessa
	\item Kolmen minuutin haudutus
	\item Huoneenlämpö ympärillä (20 C)
	\item Kolme mustaa, kolme vihreää teetä eri päivinä
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Kokeen suoritus}
	\begin{itemize}
	\item Ajastin rullaa haudutuksesta lähtien
	\item Pyöritellään mittaprobea mukissa
	\item Maistellaan \href{https://www.youtube.com/watch?v=BW1te\_miu5I\#t=1m31s}{tarkkaan}
	\item Merkitään taulukkoon aika, mitattu lämpötila ja havaittu maku
	\item Jos maku ei muuttunut aiemmasta, niin jätetään mainitsematta
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Huomioita}
	\begin{itemize}
	\item 50-60-asteinen tee on kivaa
	\item Lämpimämmällä ei ole laatua kun ei vain voi käyttää
	\item Viileämpi maistuu jotenkin huonommalta
	\item Vihreä tee ei happane niin hurjasti jäähtyessään
	\item Lämpötila mukissa vaihtelee (pinnasta haihtuu, keskellä lämpimintä)
	\item Laadun absoluuttinen määritys ei ole yhtenäistä
	\end{itemize}
